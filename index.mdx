<Steps>
  <Step title="Open the guide">
    Use the **Mintlify Guide** dropdown in the navbar, or click any card above.
  </Step>
  <Step title="Read or look up">
    Each topic is its own focused page with its own table of contents on the right.
  </Step>
  <Step title="Search anything">
    Press <kbd>Ctrl</kbd>/<kbd>⌘</kbd> + <kbd>K</kbd> to search across every page
    instantly.
  </Step>
</Steps>
